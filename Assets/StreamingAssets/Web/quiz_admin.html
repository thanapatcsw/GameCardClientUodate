# คำถาม หมวด ระดับ ตัวเลือก เฉลย

📋 รูปแบบไฟล์ JSON

ไฟล์ JSON ต้องมี structure ตามรูปแบบนี้เพื่อให้ import ได้ถูกต้อง

โครงสร้างไฟล์
{
  "version": "1.0",
  "totalQuestions": 3,
  "questions": [
    {
      "id": "q001",
      "category": "วิทยาศาสตร์",
      "difficulty": "easy",
      "question": "น้ำมีสูตรทางเคมีว่าอะไร?",
      "choices": ["H2O", "CO2", "NaCl", "O2"],
      "correctIndex": 0
    },
    {
      "id": "q002",
      "category": "คณิตศาสตร์",
      "difficulty": "medium",
      "question": "2 + 2 × 2 = ?",
      "choices": ["4", "6", "8", "2"],
      "correctIndex": 1
    }
  ]
}
ฟิลด์ที่รองรับ
Fieldประเภทบังคับรายละเอียด
idstringไม่บังคับID อ้างอิง (จะเก็บใน external_id)
categorystringไม่บังคับหมวดหมู่ (default: "ทั่วไป")
difficultystringไม่บังคับ"easy" | "medium" | "hard" (default: "medium")
questionstringบังคับข้อความคำถาม
choicesstring[]บังคับarray 4 ตัวเลือก
correctIndexnumberบังคับindex ตัวเลือกที่ถูก (0–3)
คำถามใน Patch
